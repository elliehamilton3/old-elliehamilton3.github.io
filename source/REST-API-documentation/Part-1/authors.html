    <parses>
        <list-item>application/xml</list-item>

        <list-item>text/csv</list-item>
    </parses>
    <actions>
        <POST>
            <url>
                <type>field</type>
                <required>False</required>
                <read_only>True</read_only>
                <label>Url</label>

            <name>
                <type>string</type>
